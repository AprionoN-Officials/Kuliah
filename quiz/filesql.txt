-- 1. Tabel Dept
CREATE TABLE dept (
    id int(5) NOT NULL AUTO_INCREMENT,
    dept_code varchar(10),
    deptname varchar(25),
    location varchar(50),
    PRIMARY KEY (id)
);

-- 2. Tabel Employee
CREATE TABLE employee (
    id int(5) NOT NULL AUTO_INCREMENT,
    empname varchar(50),
    phone varchar(15),
    address varchar(50),
    id_dept int(5) NOT NULL,
    PRIMARY KEY (id)
);
